🔰 Introduction à Pascal

Pascal est un langage structuré, utilisé pour enseigner les bases de la programmation (variables, conditions, boucles, procédures...)

📌 Structure d’un programme Pascal

program MonProgramme;
uses crt;

var
  x: Integer;

begin
  writeln('Bonjour Pascal');
  readln;
end.

🔠 Types de données

  • Integer : nombre entier
  • Real : nombre réel
  • Char : caractère
  • String : chaîne de caractères
  • Boolean : vrai/faux

📥 Variables & Constantes

var
  nom: String;

🔀 Conditions complètes

if a > b then
  writeln('a est plus grand')
else if a = b then
  writeln('a est égal à b')
else
  writeln('b est plus grand');

case jour of
  1: writeln('Lundi');
  2: writeln('Mardi');
  3: writeln('Mercredi');
else
  writeln('Jour inconnu');
end;

🔁 Boucles

for i := 1 to 5 do
  writeln(i);

while x < 10 do
begin
  x := x + 1;
end;

repeat
  x := x - 1;
until x = 0;

📊 Tableaux

var
  tab: array[1..5] of Integer;
  i: Integer;

begin
  for i := 1 to 5 do
    tab[i] := i * 10;

  writeln('Valeur en position 3 : ', tab[3]);
end.

📈 Tri d’un tableau (sélection)

var
  tab: array[1..5] of Integer;
  i, j, tmp: Integer;

begin
  for i := 1 to 4 do
    for j := i + 1 to 5 do
      if tab[i] > tab[j] then
      begin
        tmp := tab[i];
        tab[i] := tab[j];
        tab[j] := tmp;
      end;

  for i := 1 to 5 do
    writeln(tab[i]);
end.
